@st.cache_data(ttl=3600)  # cache for 1 hour
def get_ai_insights(prompt):

    if not client:
        return "⚠ API key not configured."

    try:
        response = client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[{"role": "user", "content": prompt}],
